## SIZE
SELECT TABLE_SCHEMA, ROUND(SUM(DATA_LENGTH + INDEX_LENGTH)/(1024*1024), 4) AS `Size (MB)` FROM information_schema.TABLES GROUP BY TABLE_SCHEMA ORDER BY
ROUND(SUM(DATA_LENGTH + INDEX_LENGTH)/(1024*1024), 4) DESC;
